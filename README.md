# 关于推送到github仓库
# 第一套指令
# 生成 .git 目录，初始化仓库
git clone https://github.com/ALight-afk/python-food-ordering-system.git
# 把github上面的仓库克隆到本地
git add .
# 把项目的所有文件提交到暂存区
git commit -m "日志信息" 
# 暂存区—>本地仓库
git push -u origin main
# 推送到github仓库
